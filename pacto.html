const conducta = JSON.parse(localStorage.getItem('conducta_elegida')); if (!conducta) { alert('No hay conducta seleccionada'); window.location.href = 'seleccion.html'; } document.getElementById('resumen').innerHTML = ` ${conducta.emoji}
${conducta.nombre}
`; document.getElementById('btnAceptar').onclick = function() { // Obtener el pacto existente si lo hay let pacto = JSON.parse(localStorage.getItem('faro_pacto')); // Intentar conservar los días ya completados si esta conducta ya existía let diasCompletadosPrevios = []; let fechasPrevias = []; if (pacto && pacto.conductasActivas) { const conductaExistente = pacto.conductasActivas.find(c => c.id === conducta.id); if (conductaExistente) { diasCompletadosPrevios = conductaExistente.diasCompletados || []; fechasPrevias = conductaExistente.diasCompletadosFecha || []; } } const nuevaConducta = { id: conducta.id, nombre: conducta.nombre, emoji: conducta.emoji, fechaInicio: new Date().toISOString(), diasCompletados: diasCompletadosPrevios, diasCompletadosFecha: fechasPrevias }; // Actualizar o crear el pacto if (!pacto) { pacto = { conductasActivas: [nuevaConducta] }; } else { const indiceExistente = pacto.conductasActivas.findIndex(c => c.id === conducta.id); if (indiceExistente !== -1) { pacto.conductasActivas[indiceExistente] = nuevaConducta; } else { pacto.conductasActivas.push(nuevaConducta); } } localStorage.setItem('faro_pacto', JSON.stringify(pacto)); alert(`¡Pacto activado para ${conducta.nombre}!`); window.location.href = 'ritmo.html'; }; document.getElementById('btnRechazar').onclick = function() { if (confirm('¿Cancelar el pacto?')) { window.location.href = 'seleccion.html'; } }; document.getElementById('btnAtras').onclick = function() { window.location.href = 'seleccion.html'; };
